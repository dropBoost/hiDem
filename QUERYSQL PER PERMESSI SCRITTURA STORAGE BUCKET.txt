create policy "upload docs (public)"
on storage.objects
for insert
to public
with check (bucket_id in ('documentiveicoli','documentidetentori'));

create policy "update docs (public)"
on storage.objects
for update
to public
using (bucket_id in ('documentiveicoli','documentidetentori'))
with check (bucket_id in ('documentiveicoli','documentidetentori'));

create policy "read docs (public)"
on storage.objects
for select
to public
using (bucket_id in ('documentiveicoli','documentidetentori'));
